#Выведите список всех файлов в папке html

ls -a var/www/html #-a чтоб увидеть скрытые файлы

#Найдите в файле access.log все упоминания строки "404"

grep --color "404" var/logs/access.log #вывожу все строки в которых есть 404 и подсвечиаю их цветом

#На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403

cp var/www/html/404.html var/www/html/403.html #копирую содержимое файла 404.html в файл 403.html
sed -i 's/404/403/g' var/www/html/403.html #меняю все 404 на 403 g-указывает что глобально, все упоминания, а не первое вхождение

#Модифицируйте права доступа и владельца файла index.html следующим образом: 
#Текущий пользователь должен иметь права на чтение/запись Группа www-data на чтение Все остальные не имеют доступа к файлу

chown :www-data var/www/html/index.html #меняю группу для файла index.html
cmod 640 var/www/html/index.html #меняю права доступа

#Сделайте символическую ссылку на файл access.log в папке html.

cd var/www/html #перехожу в папку html
ln -s var/logs/access.log mylink # -s указывает что ссылка символическая, создаю ссылку mylink

#Выведите на экран все файлы в папке var

cd ../.. #перехожу на два уровня вверх в папку var
ls -al #вывожу содержимое папки со всеми атрибутами
